# 登录

登录的功能放在 /utils/login.js文件中,如果发现本地 token过期可以直接重新调用 login 方法 获取最新的 token

const { put, get, remove } = require("./redis");
const { login: loginApi } = require("./service");
import to from "await-to-js";
const app = getApp();
module.exports = {
  login() {
    async function loginRequest(code) {
      wx.showLoading({
        title: "加载中",
      });
      const [errMsg, { data }] = await to(loginApi(code));
      wx.hideLoading();
      if (errMsg) {
        wx.showToast({
          title: "接口请求失败",
          icon: "error",
        });
      } else {
        if (data?.retCode !== 200) {
          wx.showToast({
            title: data?.message ?? data?.msg,
            icon: "none",
            duration: 2000,
          });
        } else {
          put("openId", data.data);
          remove("code");
        }
      }
    }
    const openid = app.getOpenId();
    if (!openid) {
      const code = get("code");
      if (code) {
        loginRequest(code);
      } else {
        wx.login({
          success: ({ code, errMsg }) => {
            if (errMsg === "login:ok") {
              //缓存code 4分钟 防止获取openid失败 不停的调用code导致 wx.login 调用上限
              put("code", code, 240);
              loginRequest(code);
            } else {
              wx.showToast({
                title: errMsg,
              });
            }
          },
        });
      }
    }
  },
};

# 存储

本地存储的封装放在了 /utils/redis.js文件中，常用操作进行了封装，支持设置过期时间。

var redis = "redis";

/**
 * 设置
 * k 键key
 * v 值value
 * t 秒
 */
function put(k, v, t = 0) {
  wx.setStorageSync(k, v);
  var seconds = parseInt(t);
  if (seconds > 0) {
    var newtime = Date.parse(new Date());
    newtime = newtime / 1000 + seconds;
    wx.setStorageSync(k + redis, newtime + "");
  } else {
    wx.removeStorageSync(k + redis);
  }
}
/**
 * 获取
 * k 键key
 */
function get(k) {
  var deadtime = parseInt(wx.getStorageSync(k + redis));
  if (deadtime) {
    if (parseInt(deadtime) < Date.parse(new Date()) / 1000) {
      wx.removeStorageSync(k);
      console.log("过期了");
      return null;
    }
  }
  var res = wx.getStorageSync(k);
  if (res) {
    return res;
  } else {
    return null;
  }
}

/**
 * 删除
 */
function remove(k) {
  wx.removeStorageSync(k);
  wx.removeStorageSync(k + redis);
}

/**
 * 清除所有key
 */
function clear() {
  wx.clearStorageSync();
}
module.exports = {
  put,
  get,
  remove,
  clear,
};

# store

之前是把openId存在了本地的Storage中 现在把openId存入到了全局的store 使用了wxministore准备包

// app.js
const Store = require('wxministore');
let store = new Store({
    state: {
        openId: ''
    }
})
App({
    onLaunch() {
        let system = wx.getSystemInfoSync()
        //苹果X底部安全区域
        let bottomSafe = system.screenHeight - system.safeArea.bottom
        //减去头部区域和底部安全区域
        this.globalData.windowHeight = system.windowHeight - bottomSafe - 85

    },
    store: store,
    globalData: {
        //小程序页面高度
        windowHeight: 0,
    },
})
