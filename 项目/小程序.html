# 登录

登录的功能放在 /utils/login.js文件中,如果发现本地 token过期可以直接重新调用 login 方法 获取最新的 token

const { put, get, remove } = require("./redis");
const { login: loginApi } = require("./service");
const clear = require("./clear");

import to from "await-to-js";
const app = getApp();
/**
 * 调用登录API
 * @param {wx.login返回的code} code
 */
const loginRequest = async (code) => {
    const [, result] = await to(loginApi(code));
    const data = result?.data;
    if (data?.retCode !== 200) {
        // const { loginErrorCount } = app.store.getState();
        // //重试3次
        // if (loginErrorCount < 2) {
        //     app.store.setState({
        //         loginErrorCount: loginErrorCount + 1,
        //     });
        //     //删除code
        //     remove("code");
        //     wx.reLaunch({
        //         url: "/pages/index/index",
        //     });
        // }
    } else {
        put("openId", data.data.token, 86400);
        app.store.setState({
            token: data.data.token,
        });
        remove("code");
    }
};
/**
 * 登录业务的封装
 */
const handleLogin = () => {
    const code = get("code");
    if (code) {
        loginRequest(code);
        return;
    }
    wx.login({
        success: ({ code, errMsg }) => {
            if (errMsg === "login:ok") {
                //缓存code 4分钟 防止获取openid失败 不停的调用code导致 wx.login 调用上限
                put("code", code, 240);
                loginRequest(code);
                return;
            }
            wx.showToast({
                title: errMsg,
            });
        },
    });
};
module.exports = {
    /**
     * 对外暴露的登录方法
     */
    login() {
        const { token } = app.store.getState();
        if (!token) handleLogin();
    },
};

# 存储

本地存储的封装放在了 /utils/redis.js文件中，常用操作进行了封装，支持设置过期时间。

var redis = "redis";

/**
 * 设置
 * k 键key
 * v 值value
 * t 秒
 */
function put(k, v, t = 0) {
  wx.setStorageSync(k, v);
  var seconds = parseInt(t);
  if (seconds > 0) {
    var newtime = Date.parse(new Date());
    newtime = newtime / 1000 + seconds;
    wx.setStorageSync(k + redis, newtime + "");
  } else {
    wx.removeStorageSync(k + redis);
  }
}
/**
 * 获取
 * k 键key
 */
function get(k) {
  var deadtime = parseInt(wx.getStorageSync(k + redis));
  if (deadtime) {
    if (parseInt(deadtime) < Date.parse(new Date()) / 1000) {
      wx.removeStorageSync(k);
      console.log("过期了");
      return null;
    }
  }
  var res = wx.getStorageSync(k);
  if (res) {
    return res;
  } else {
    return null;
  }
}

/**
 * 删除
 */
function remove(k) {
  wx.removeStorageSync(k);
  wx.removeStorageSync(k + redis);
}

/**
 * 清除所有key
 */
function clear() {
  wx.clearStorageSync();
}
module.exports = {
  put,
  get,
  remove,
  clear,
};

# store

之前是把openId存在了本地的Storage中 现在把openId存入到了全局的store 使用了wxministore

// store/store.js
const { get } = require("../utils/redis");
const Store = require("wxministore");
module.exports = new Store({
    state: {
        //登录的token
        token: get("openId") ?? "",
        //切库的token
        authorization: "",
        //首页选择的项目(选择的缴费单位) 全部使用的是tenantId字段
        selectedProjectID: 0,
        //登录错误的次数 只会重试3次
        loginErrorCount: 0,
    },
});
