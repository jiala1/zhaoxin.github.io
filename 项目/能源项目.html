utils/http/index.js

//开发环境使用
if (import.meta.env.DEV) {
return `/${prefix}/${url}`;
} else {
// let port = 80;
//生产环境 配置在.env文件中
// const httpPrefixPort = JSON.parse(import.meta.env.VITE_HTTP_PREFIX_PORT);
const httpPrefix = JSON.parse(import.meta.env.VITE_HTTP_APACHE_PREFIX);
//apache 前缀
let apacheForwardingPrefix: string;
switch (prefix) {
  case "system":
    // port = httpPrefixPort["system"];
    apacheForwardingPrefix = httpPrefix["system"];
    break;
  case "energy":
    // port = httpPrefixPort["energy"];
    apacheForwardingPrefix = httpPrefix["energy"];
    break;
  case "router":
    // port = httpPrefixPort["router"];
    apacheForwardingPrefix = httpPrefix["router"];
    break;
  case "pmp":
    // port = httpPrefixPort["pmp"];
    apacheForwardingPrefix = httpPrefix["pmp"];
    break;
  default:
    throw new Error("Invalid prefix");
}
//return `${import.meta.env.VITE_HTTP_URL}:${port}/${url}`;
return `/${apacheForwardingPrefix}/${url}`;

# excel导出

hooks目录下 useExcelExporter.ts

/**
 * 封装excel导出hooks
 * @param column excel字段
 * @param tableData 表格字段
 * @param excelName 表格名称
 * @param config 配置
 * @param exportExcelLoading 导出loading状态
 */
export default function (
  column: Array<{ title: string; dataIndex: string }>,
  tableData: Array<any>,
  excelName: string,
  config: object = {},
  exportExcelLoading?: Ref<boolean>
) {
  const progress = ref(0);
  watch(progress, newVal => {
    if (newVal === 100) {
      exportExcelLoading && (exportExcelLoading.value = false);
    }
  });
  exportExcelLoading && (exportExcelLoading.value = true);
  //导出excel
  exportExcelComplex(column, cloneDeep(tableData), excelName, config, progress);
  return { progress };
}
